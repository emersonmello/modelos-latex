\question[10]
Calcule $\displaystyle\int_0^1 x \, dx$.

\begin{solutionorbox}[2cm]
$\displaystyle\tfrac{x^2}{2}\Big|^1_0 = \tfrac{1^2}{2} - \tfrac{0^2}{2} = \tfrac{1}{2}$
\end{solutionorbox}
